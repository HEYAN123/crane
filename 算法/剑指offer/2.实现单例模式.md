# 实现单例模式

## 题目

- 设计一个类，我们只能生成该类的一个实例。

## 解析

- 核心思想：用一个变量来标志当前是否已经为某个类创建过对象，如果是，则下次获取该类的实例时直接返回之前创建的对象

```javascript
// 单例类
var Singleton = function(name) {
    this.name = name;
}

Singleton.prototype.getName = function() {
    alert(this.name);
}

Singleton.getInstance = (function(){
    var instance = null;
    return function(name) {
        if(!instance) {
            instance = new Singleton(name);
        }
        return instance;
    }
})();

// ----------------闭包实现单例
var CreateDiv = (function() {
    var instance;
    // 创建对象
    var CreateDiv = function(html) {
        if(instance) {
            return instance;
        }
        this.html = html;
        this.init();
        return instance = this;
    };
    // 执行初始化
    CreateDiv.prototype.init = function() {
        var div = document.createElement('div');
        div.innerHTML = this.html;
        document.body.appendChild(div);
    };
    return CreateDiv;
})();

var a = new CreateDiv('111');
var b = new CreateDiv('222');
console.log(a === b); // true

// -------------单一职责原则
// 只负责构造对象
var CreateDiv = function(html) {
    this.html = html;
    this.init();
}

CreateDiv.prototype.init = function () {
    var div = document.createElement('div');
    div.innerHTML = this.html;
    document.body.appendChild(div);
}
// 判断返回新对象还是旧对象
var ProxySingletonCreateDiv = (function() {
    var instance;
    return function(html) {
        if(!instance) {
            instance = new CreateDiv(html);
        }
        return instance;
    }
})();

var a = new ProxySingletonCreateDiv('111');
var b = new ProxySingletonCreateDiv('222');
console.log(a === b); // true

// 单例模式代码
// 一个构造函数
var Singleton = function(name) {
    this.name = name;
}

Singleton.getInstance = (function(name){
    var instance;
    return function(name) {
        if(!instance) {
            instance = new Singleton(name);
        }
        return instance;
    }
})(); // 立即执行，得到闭包式构造函数

var a = Singleton.getInstance('111');
var b = Singleton.getInstance('222');
console.log(a===b, a, b);

// 抽象代码
var getSingle = function(fn) {
    var result;
    return function() {
        if(!result) result = new fn(...arguments);
        return result;
    }
}

var Person = function(name) {
    this.name = name;
}

var SinglePerson = getSingle(Person);

var a = SinglePerson('ann');
var b = SinglePerson('tom');
console.log(a===b, a, b);

```
